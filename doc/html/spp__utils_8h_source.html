172 inline size_t spp_mix_32(uint32_t a)
173 {
174  a = a ^ (a >> 4);
175  a = (a ^ 0xdeadbeef) + (a << 5);
176  a = a ^ (a >> 11);
177  return static_cast<size_t>(a);
178 }

320 static inline uint32_t s_spp_popcount_default(uint32_t i) SPP_NOEXCEPT
321 {
322  i = i - ((i >> 1) & 0x55555555);
323  i = (i & 0x33333333) + ((i >> 2) & 0x33333333);
324  return (((i + (i >> 4)) & 0x0F0F0F0F) * 0x01010101) >> 24;
325 }
326 
327 static inline uint32_t s_spp_popcount_default(uint64_t x) SPP_NOEXCEPT
328 {
329  const uint64_t m1 = uint64_t(0x5555555555555555); // binary: 0101...
330  const uint64_t m2 = uint64_t(0x3333333333333333); // binary: 00110011..
331  const uint64_t m4 = uint64_t(0x0f0f0f0f0f0f0f0f); // binary: 4 zeros, 4 ones ...
332  const uint64_t h01 = uint64_t(0x0101010101010101); // the sum of 256 to the power of 0,1,2,3...
333 
334  x -= (x >> 1) & m1; // put count of each 2 bits into those 2 bits
335  x = (x & m2) + ((x >> 2) & m2); // put count of each 4 bits into those 4 bits
336  x = (x + (x >> 4)) & m4; // put count of each 8 bits into those 8 bits
337  return (x * h01)>>56; // returns left 8 bits of x + (x<<8) + (x<<16) + (x<<24)+...
338 }
